namespace Bondstone.Capabilities.DomainEvents
{
    [System.AttributeUsage(System.AttributeTargets.Class | System.AttributeTargets.Struct, Inherited=false)]
    public sealed class DomainEventIdentityAttribute : System.Attribute
    {
        public DomainEventIdentityAttribute(string name) { }
        public string Name { get; }
    }
    public interface IDomainEvent { }
    public interface IDomainEventHandler<in TDomainEvent>
        where in TDomainEvent : Bondstone.Capabilities.DomainEvents.IDomainEvent
    {
        System.Threading.Tasks.ValueTask HandleAsync(TDomainEvent domainEvent, System.Threading.CancellationToken ct = default);
    }
    public interface IDomainEventSource
    {
        System.Collections.Generic.IReadOnlyCollection<Bondstone.Capabilities.DomainEvents.IDomainEvent> PendingDomainEvents { get; }
        void ClearPendingDomainEvents();
    }
}
