namespace Bondstone.Configuration
{
    public sealed class DurableTransportTopologyRouteDiagnostic
    {
        public DurableTransportTopologyRouteDiagnostic(string transportName, Bondstone.Messaging.MessageKind messageKind, string routeSubject, bool hasRoute, string? failureReason = null) { }
        public string? FailureReason { get; }
        public bool HasRoute { get; }
        public Bondstone.Messaging.MessageKind MessageKind { get; }
        public string RouteSubject { get; }
        public string TransportName { get; }
    }
    public interface IDurableTransportTopologyDiagnosticSource
    {
        string TransportName { get; }
        Bondstone.Configuration.DurableTransportTopologyRouteDiagnostic DescribeCommandRoute(string targetModule);
        Bondstone.Configuration.DurableTransportTopologyRouteDiagnostic DescribeEventRoute(string messageTypeName);
    }
}
namespace Bondstone.Messaging
{
    public enum DurableMessageTopologyDiagnosticKind
    {
        CommandRoute = 1,
        CommandDestination = 2,
        CommandReceiveEndpoint = 3,
        EventDestination = 4,
        EventSubscription = 5,
    }
}
